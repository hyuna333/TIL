- stack
  ```python
  class Stack:
      def __init__(self):
          self.stc = []

      def empty(self):
          if self.stc == []:
              return True
          else:
              return False
      
      def top(self):
          return self.stc[-1]

      def pop(self):
          if self.stc == []:
              return None
          else:
              print(self.stc.pop(-1))
      
      def push(self, x):
          self.stc.append(x)
      
      def __repr__(self):
          return self.stc
  ```
